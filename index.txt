    <script>
        // 1. Firebase 설정 (반드시 본인의 정보로 입력하세요)
        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            databaseURL: "YOUR_DATABASE_URL",
            projectId: "YOUR_PROJECT_ID",
        };
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        // 2. 보드 데이터 구성
        const BOARD_DATA = {};
        const cityNames = ["출발", "서울", "부산", "도쿄", "베이징", "무인도", "런던", "파리", "로마", "베를린", "사회복지", "뉴욕", "시드니", "방콕", "제주", "우주여행", "모스크바", "델리", "카이로", "상파울루"];
        // 0~39번 칸 생성 (간략화된 예시 데이터)
        for(let i=0; i<40; i++) {
            BOARD_DATA[i] = {
                name: cityNames[i % cityNames.length],
                price: (i === 0) ? 0 : 50 + (i * 5),
                build: 30 + (i * 2),
                rent: [10 + i, 50 + i, 150 + i, 500 + i] // 대지, 집, 빌딩, 호텔 순
            };
        }

        // 3. 좌표 시스템 (0~39 시계방향)
        function getCoords(pos) {
            if (pos <= 10) return { r: 11, c: 11 - pos }; // 하단 (오른쪽->왼쪽)
            if (pos <= 20) return { r: 11 - (pos - 10), c: 1 }; // 왼쪽 (아래->위)
            if (pos <= 30) return { r: 1, c: pos - 20 + 1 }; // 상단 (왼쪽->오른쪽)
            return { r: pos - 30 + 1, c: 11 }; // 오른쪽 (위->아래)
        }

        class GameEngine {
            constructor(roomId, myId) {
                this.roomId = roomId;
                this.myId = myId;
                this.roomRef = db.ref(`rooms/${roomId}`);
                this.playerColor = myId === 'p1' ? 'red' : 'blue';
                this.initBoard();
                this.syncData();
            }

            initBoard() {
                const board = document.getElementById('game-board');
                for (let i = 0; i < 40; i++) {
                    const coords = getCoords(i);
                    const tileDiv = document.createElement('div');
                    tileDiv.className = 'tile';
                    tileDiv.id = `tile-${i}`;
                    tileDiv.style.gridRow = coords.r;
                    tileDiv.style.gridColumn = coords.c;
                    tileDiv.innerHTML = `<div>${BOARD_DATA[i].name}</div>`;
                    if(BOARD_DATA[i].price > 0) {
                        tileDiv.innerHTML += `<div class="price">${BOARD_DATA[i].price}</div>`;
                    }
                    board.appendChild(tileDiv);
                }
            }

            syncData() {
                this.roomRef.on('value', (snap) => {
                    const data = snap.val();
                    if (!data) return;

                    // 플레이어 렌더링
                    Object.keys(data.players).forEach(pid => {
                        this.updateToken(pid, data.players[pid]);
                    });

                    // 내 정보 업데이트
                    const me = data.players[this.myId];
                    document.getElementById('my-money').innerText = me.money;
                    document.getElementById('current-turn-name').innerText = data.turn === this.myId ? "내 차례!" : "상대방 차례";
                    document.getElementById('roll-btn').disabled = (data.turn !== this.myId);

                    // 타일 소유주 표시
                    if(data.tiles) {
                        Object.keys(data.tiles).forEach(tIdx => {
                            const tInfo = data.tiles[tIdx];
                            const tileEl = document.getElementById(`tile-${tIdx}`);
                            tileEl.style.backgroundColor = tInfo.owner === 'p1' ? '#ffcdd2' : '#bbdefb';
                            tileEl.querySelector('.price').innerText = `Lv.${tInfo.level}`;
                        });
                    }
                });
            }

            updateToken(pid, pData) {
                let token = document.getElementById(`token-${pid}`);
                if (!token) {
                    token = document.createElement('div');
                    token.id = `token-${pid}`;
                    token.className = 'player-token';
                    token.style.backgroundColor = pid === 'p1' ? 'red' : 'blue';
                    token.innerText = pid;
                    document.getElementById('game-board').appendChild(token);
                }
                const coords = getCoords(pData.position);
                token.style.gridRow = coords.r;
                token.style.gridColumn = coords.c;
                // 말이 겹치지 않게 살짝 오프셋
                token.style.transform = pid === 'p1' ? 'translate(-5px, -5px)' : 'translate(10px, 10px)';
            }

            async rollDice() {
                const dice = Math.floor(Math.random() * 6) + 1;
                this.addLog(`🎲 주사위 ${dice}가 나왔습니다!`);

                const snap = await this.roomRef.once('value');
                const data = snap.val();
                let newPos = (data.players[this.myId].position + dice) % 40;
                let moneyPlus = (data.players[this.myId].position + dice >= 40) ? 20 : 0;

                // 위치 및 자산 업데이트
                await this.roomRef.child(`players/${this.myId}`).update({
                    position: newPos,
                    money: data.players[this.myId].money + moneyPlus
                });

                if(moneyPlus > 0) this.addLog("출발지를 통과하여 20만 원을 받았습니다.");

                this.handleTileAction(newPos);
            }

            async handleTileAction(pos) {
                if (pos === 0) return this.endTurn(); // 출발지
                
                const tileData = BOARD_DATA[pos];
                const snap = await this.roomRef.child(`tiles/${pos}`).once('value');
                const tileOwnerInfo = snap.val();

                if (!tileOwnerInfo) {
                    // 빈 땅 구매 확인
                    if (confirm(`${tileData.name} (가격: ${tileData.price})을 구매하시겠습니까?`)) {
                        this.processTransaction(pos, tileData.price, 'buy');
                    } else {
                        this.endTurn();
                    }
                } else if (tileOwnerInfo.owner !== this.myId) {
                    // 통행료 지불
                    const rent = tileData.rent[tileOwnerInfo.level];
                    alert(`${tileOwnerInfo.owner}의 땅입니다! 통행료 ${rent}만 원을 지불합니다.`);
                    this.processTransaction(pos, rent, 'rent', tileOwnerInfo.owner);
                } else {
                    // 내 땅 - 업그레이드
                    if (tileOwnerInfo.level < 3 && confirm(`건물을 추가하시겠습니까? (비용: ${tileData.build})`)) {
                        this.processTransaction(pos, tileData.build, 'upgrade');
                    } else {
                        this.endTurn();
                    }
                }
            }

            async processTransaction(pos, amount, type, receiverId = null) {
                const nextTurn = this.myId === 'p1' ? 'p2' : 'p1';
                
                await this.roomRef.transaction((state) => {
                    if (state) {
                        const myMoney = state.players[this.myId].money;
                        if (myMoney < amount) {
                            alert("잔액이 부족합니다!");
                            return;
                        }
                        
                        state.players[this.myId].money -= amount;
                        
                        if (type === 'buy') {
                            if (!state.tiles) state.tiles = {};
                            state.tiles[pos] = { owner: this.myId, level: 0 };
                        } else if (type === 'upgrade') {
                            state.tiles[pos].level += 1;
                        } else if (type === 'rent') {
                            state.players[receiverId].money += amount;
                        }
                        
                        state.turn = nextTurn; // 턴 넘기기
                    }
                    return state;
                });
                this.addLog(`${type === 'rent' ? '통행료 지불' : '거래'} 완료.`);
            }

            endTurn() {
                const nextTurn = this.myId === 'p1' ? 'p2' : 'p1';
                this.roomRef.update({ turn: nextTurn });
            }

            addLog(msg) {
                const win = document.getElementById('log-window');
                win.innerHTML += `<div>- ${msg}</div>`;
                win.scrollTop = win.scrollHeight;
            }
        }

        // 초기 데이터 세팅 (방이 없을 경우)

                    p2: { position: 0, money: 1500 }
                }
            });
        }

        // initGameRoom(myRoomId); // 처음 한 번만 실행해서 DB 초기화 필요
        const engine = new GameEngine(myRoomId, 'p1');
        document.getElementById('roll-btn').onclick = () => engine.rollDice();
